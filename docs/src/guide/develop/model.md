```go
func (l *DownloadLogic) Download(req *types.DownloadRequest) error {
    l.Logger.Infof("download req: %v", req)

    body, err := os.ReadFile(filepath.Join("./filedata", req.File))
    if err != nil {
        return err
    }

    n, err := l.writer.Write(body)
    if err != nil {
        return err
    }

    if n < len(body) {
        return io.ErrClosedPipe
    }

    return nil
}
```
